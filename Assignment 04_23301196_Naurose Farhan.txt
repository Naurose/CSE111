public class Bird {
    public String name;
    public int height;

    public void makeNoise() {
        if (name.equals("Parrot")) {
            System.out.println("Squawk");
        }
        else if (name.equals("Eagle")) {
            System.out.println("Squee");
        }
    }

    public void flyUp(int height){
        this.height += height;
        System.out.println(name + " has flown up " + height + " feet.");
    }

    public void flyDown(int height){
        if (this.height - height == 0){
            this.height -= height;
            System.out.println(name + " has flown down " + height + " feet and landed.");
        }
        else if (this.height - height > 0){
            this.height -= height;
            System.out.println(name + " has flown down " + height + " feet.");
        }
        else{
            System.out.println(name + " cannot fly down " + height + " feet.");
        }
    }
}



// TASK 9
public class CellPhone {
    public String model = "Unknown";
    public String [] store = new String[3];
    public int contacts;

    public void storeContact(String names){
        if (contacts < 3){
            System.out.println("Contact Stored");
            store[contacts] = names;
            contacts += 1;
        }
        else{
            System.out.println("Memory full. New contact can't be stored.");
        }
    }
    public void printDetails() {
        System.out.println("Phone Model " + model);
        System.out.println("Contact Stored " + contacts);
        if (contacts > 0){
            System.out.println("Stored Contacts:");
            for(int i = 0; i < contacts; i+=1){
                System.out.println(store [i]);
            }
        }
    }
}
