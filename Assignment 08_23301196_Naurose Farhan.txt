public class Book{
    public String name;
    public double price = 150;
    
    public static int total_books_sold;
    public static double total_revenue;

    public Book(String name, int discount){
        this.name = name;
        this.price = Discount(discount);
        total_books_sold++;
        total_revenue += this.price;
    }

    public double Discount(int discount){
        double price = this.price - (this.price * discount/100);
        return price;
    }

    public void bookDetails(){
        System.out.println("Title: " + this.name);
        System.out.println("Price after Discount: " + this.price + " TK");
    }
}

public class Borrower{
    public static int book_count[] = {3, 3, 3};
    public static String book_name[] = {"Pather Panchali", "Durgesh Nandini", "Anandmath"};
    public String name;
    public String[] arr = new String[3];
    public int index = 0;
    
    public Borrower(String name){
      this.name = name; 
    }
    
    public static void bookStatus(){
      System.out.println("Available Books:"); 
      for(int i = 0; i < book_name.length; i++){
        System.out.println(book_name[i] + ": " + book_count[i]);
      }
    }
    
    public void borrowBook(String b){
      if(remainingBooks(b) > 0){
        for(int i = 0; i < book_name.length; i++){
          if(b.equals(book_name[i])){
            book_count[i]--; 
          }
        }
        arr[index] = b;
        index++;
      }
      else{
        System.out.println("This book is not available.");
      }
    }
    
    public void borrowerDetails(){
      System.out.println("Name: " + name + "\nBooks Borrowed:");
      for(int i = 0; i < index; i++){
        System.out.println(arr[i]);
      }
    } 
    
    public static int remainingBooks(String b){
      for(int i = 0; i < book_name.length; i++){
        if(book_name[i].equals(b)){
          return book_count[i]; 
        }
      }   
      return -1;
    }
  }

public class Cargo{
    public int id;
    public String contents;
    public double weight;
    public boolean load;
    
    public static double totalCapacity = 10.0;
    public static int totalCargo = 0;

    public Cargo(String contents, double weight){
        this.contents = contents;
        this.weight = weight;
        this.load = false;
        totalCargo++;
        this.id = totalCargo;
    }

    public void load(){
        if(!load && totalCapacity >= weight){
            load = true;
            totalCapacity -= weight;
            System.out.println("Cargo " + id + " loaded for transport.");
        }
        else if(load){
            System.out.println("Cargo " + id + " is already loaded.");
        }
        else{
            System.out.println("Cannot load cargo, exceeds weight capacity.");
        }
    }

    public void unload(){
        if(load){
            load = false;
            totalCapacity += weight;
            System.out.println("Cargo " + id + " unloaded.");
        }
        else{
            System.out.println("Cargo " + id + " is not loaded.");
        }
    }

    public void details(){
        System.out.println("Cargo ID: " + id + ", Contents: " + contents + ", Weight: " + weight + ", Loaded: " + load);
    }

    public static double capacity(){
        return totalCapacity;
    }

}
